```c#
	/// <summary>
    /// 构造查询类方法返回数据列表的数量，当前页等
    /// </summary>
    /// <typeparam name="T"></typeparam>
    public class ExecuteOutParam<T> : List<T>
    {
        public int CurrentPage { get; private set; }
        public int TotalPages { get; private set; }
        public int PageSize { get; private set; }
        public int TotalCount { get; private set; }


        public ExecuteOutParam(List<T> items, int count, int pageNumber, int pageSize)
        {
            TotalCount = count;
            PageSize = pageSize;
            CurrentPage = pageNumber;
            TotalPages = (int)Math.Ceiling(count / (double)pageSize);
            AddRange(items);
        }

        public static async Task<ExecuteOutParam<T>> CreateAsync(IQueryable<T> source, int pageNumber, int pageSize)
        {
            var count = await source.CountAsync();
            var items = await source.Skip((pageNumber - 1) * pageSize).Take(pageSize).ToListAsync();
            return new ExecuteOutParam<T>(items, count, pageNumber, pageSize);
        }
    }
```
